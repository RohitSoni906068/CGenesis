```c

    printf("Result: %.2f\n", number * 2);
```

### ğŸ”¹ Pattern 2: Mathematical Calculations

```c

    float radius = 5.0;
    float area = PI * radius * radius;
    printf("Area: %.2f\n", area);
    return 0;
}
```

### ğŸ”¹ Pattern 3: Multiple Inputs

```c

    printf("Enter breadth: ");
    scanf("%f", &breadth);
    printf("Area: %.2f\n", length * breadth);
    return 0;
}
```
